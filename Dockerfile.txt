# 1. Imagem Base (Node 20 no Linux Debian Slim)
FROM node:20-slim

# 2. Instalação de Dependências do Sistema (Obrigatório para Baileys, Sharp e Puppeteer)
# Mantendo TODAS as bibliotecas da sua versão anterior para garantir compatibilidade total.
RUN apt-get update && apt-get install -y \
    ffmpeg \
    libvips-dev \
    ca-certificates \
    chromium \
    fonts-liberation \
    libasound2 \
    libatk-bridge2.0-0 \
    libatk1.0-0 \
    libc6 \
    libcairo2 \
    libcups2 \
    libdbus-1-3 \
    libexpat1 \
    libfontconfig1 \
    libgbm1 \
    libgcc1 \
    libglib2.0-0 \
    libgtk-3-0 \
    libnspr4 \
    libnss3 \
    libpango-1.0-0 \
    libpangocairo-1.0-0 \
    libstdc++6 \
    libx11-6 \
    libx11-xcb1 \
    libxcomposite1 \
    libxcursor1 \
    libxdamage1 \
    libxext6 \
    libxfixes3 \
    libxi6 \
    libxrandr2 \
    libxrender1 \
    libxss1 \
    libxtst6 \
    lsb-release \
    wget \
    xdg-utils \
    curl \
    --no-install-recommends \
    && rm -rf /var/lib/apt/lists/*

# 3. Diretório de Trabalho
WORKDIR /app

# 4. Cache de Dependências
COPY package*.json ./

# 5. Instalação das Dependências Node.js
RUN npm ci --production

# 6. Copia o código fonte
COPY . .

# 7. Configurações de Ambiente
ENV NODE_ENV=production
# CORREÇÃO: Porta 3000 para alinhar com o server.js
ENV PORT=3000
ENV PUPPETEER_SKIP_CHROMIUM_DOWNLOAD=true
ENV PUPPETEER_EXECUTABLE_PATH=/usr/bin/chromium
# Otimização de Memória para instâncias robustas
ENV NODE_OPTIONS="--max-old-space-size=4096"

# 8. Expõe a porta da API
EXPOSE 3000

# 9. Comando de Inicialização
CMD ["node", "server.js"]
