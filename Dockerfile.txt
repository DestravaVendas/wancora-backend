# 1. Imagem Base (Node 20 no Linux Debian Slim - Estabilidade e Performance)
FROM node:20-slim

# 2. Instalação de Dependências do Sistema (Obrigatório para Baileys, Sharp e Puppeteer)
# ffmpeg: Conversão de áudio/vídeo.
# libvips-dev: Processamento de imagem ultra-rápido (Sharp).
# chromium e libs: Garante que o spoofing de navegador funcione sem crashar.
RUN apt-get update && apt-get install -y \
    ffmpeg \
    libvips-dev \
    ca-certificates \
    chromium \
    fonts-liberation \
    libasound2 \
    libatk-bridge2.0-0 \
    libatk1.0-0 \
    libc6 \
    libcairo2 \
    libcups2 \
    libdbus-1-3 \
    libexpat1 \
    libfontconfig1 \
    libgbm1 \
    libgcc1 \
    libglib2.0-0 \
    libgtk-3-0 \
    libnspr4 \
    libnss3 \
    libpango-1.0-0 \
    libpangocairo-1.0-0 \
    libstdc++6 \
    libx11-6 \
    libx11-xcb1 \
    libxcb1 \
    libxcomposite1 \
    libxcursor1 \
    libxdamage1 \
    libxext6 \
    libxfixes3 \
    libxi6 \
    libxrandr2 \
    libxrender1 \
    libxss1 \
    libxtst6 \
    lsb-release \
    wget \
    xdg-utils \
    curl \
    --no-install-recommends \
    && rm -rf /var/lib/apt/lists/*

# 3. Definição do Diretório de Trabalho
WORKDIR /app

# 4. Copia arquivos de dependência (Cache Layering)
COPY package*.json ./

# 5. Instalação das Dependências Node.js
# --production: Pula devDependencies para economizar espaço e segurança
RUN npm ci --production

# 6. Copia o restante do código fonte
COPY . .

# 7. Configurações de Ambiente (CORRIGIDAS)
ENV NODE_ENV=production
# IMPORTANTE: A porta deve ser 3000 para alinhar com o server.js
ENV PORT=3000
ENV PUPPETEER_SKIP_CHROMIUM_DOWNLOAD=true
ENV PUPPETEER_EXECUTABLE_PATH=/usr/bin/chromium
# Otimização de Memória para evitar crash em instâncias pesadas
ENV NODE_OPTIONS="--max-old-space-size=4096"

# 8. Expõe a porta da API (Corrigida para 3000)
EXPOSE 3000

# 9. Comando de Inicialização
# Usamos node direto para maior controle de sinais do sistema (SIGTERM/SIGINT)
CMD ["node", "server.js"]
